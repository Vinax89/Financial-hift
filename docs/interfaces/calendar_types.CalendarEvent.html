    Interface CalendarEvent

    Represents a financial event on the calendar

    Remarks

    A calendar event can represent various types of financial activities including income, expenses, bills, goals, or work shifts. Events are displayed on the calendar view with appropriate visual styling based on their type.

    Example

    const event: CalendarEvent = {
      id: '123',
      title: 'Rent Payment',
      amount: 1500.00,
      date: new Date('2025-10-01'),
      type: 'bill',
      category: 'Housing',
      status: 'pending'
    };
    interface CalendarEvent {
        id: string | number;
        title: string;
        amount: number;
        date: string | Date;
        type: "income" | "expense" | "shift" | "bill" | "goal";
        category?: string;
